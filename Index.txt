function fmt(n) {
  return n != null ? "$" + Math.round(n).toLocaleString("es-AR") : "—";
}

function relTime(ts) {
  if (!ts) return null;
  const d = new Date(ts.replace(" ","T")), now = new Date();
  const mins = Math.round((now - d) / 60000);
  if (mins < 60) return `hace ${mins}m`;
  if (mins < 1440) return `hace ${Math.round(mins/60)}h`;
  return `hace ${Math.round(mins/1440)}d`;
}

function fmtFecha(ts) {
  if (!ts) return "";
  const d = new Date(ts.replace(" ","T"));
  return d.toLocaleDateString("es-AR", { day:"2-digit", month:"2-digit" });
}

function promoColor(promo) {
  if (!promo) return "#1565C0";
  if (/\dx\d/.test(promo)) return "#FF6B00";
  if (promo.includes("unidad")) return "#7B1FA2";
  return "#C62828";
}

// Sparkline component para mini gráfico en tarjeta
function Sparkline({ data }) {
  if (!data || data.length < 2) return null;
  const width = 100;
  const height = 30;
  const prices = data.map(d => d.precio);
  const min = Math.min(...prices);
  const max = Math.max(...prices);
  const range = max - min || 1;
  
  const points = data.map((d, i) => {
    const x = (i / (data.length - 1)) * width;
    const y = height - ((d.precio - min) / range) * height;
    return `${x},${y}`;
  }).join(" ");
  
  return (
    <svg className="sparkline" viewBox={`0 0 ${width} ${height}`} preserveAspectRatio="none">
      <polyline
        points={points}
        fill="none"
        stroke="#E91E63"
        strokeWidth="2"
        strokeLinecap="round"
        strokeLinejoin="round"
      />
    </svg>
  );
}

function MiniChart({ data }) {
  if (!data || data.length < 2) return null;
  const W = 460, H = 140, PAD = { top:10, right:10, bottom:24, left:60 };
  const innerW = W - PAD.left - PAD.right;
  const innerH = H - PAD.top - PAD.bottom;
  const precios = data.map(d => d.precio);
  const minP = Math.min(...precios);
  const maxP = Math.max(...precios);
  const rng = maxP - minP || 1;
  const xStep = innerW / (data.length - 1);
  const toX = i => PAD.left + i * xStep;
  const toY = v => PAD.top + innerH - ((v - minP) / rng) * innerH;
  const points = data.map((d,i) => `${toX(i)},${toY(d.precio)}`).join(" ");
  const areaPoints = [
    `${toX(0)},${PAD.top + innerH}`,
    ...data.map((d,i) => `${toX(i)},${toY(d.precio)}`),
    `${toX(data.length-1)},${PAD.top + innerH}`
  ].join(" ");

  return (
    <svg viewBox={`0 0 ${W} ${H}`} style={{width:"100%",height:H}}>
      {[0,.5,1].map(t => {
        const y = PAD.top + innerH * (1 - t);
        const val = minP + rng * t;
        return (
          <g key={t}>
            <line x1={PAD.left} y1={y} x2={W-PAD.right} y2={y}
              stroke="var(--border)" strokeWidth="1" />
            <text x={PAD.left - 6} y={y+4} textAnchor="end"
              fontSize="10" fill="var(--text-secondary)">
              ${Math.round(val).toLocaleString("es-AR")}
            </text>
          </g>
        );
      })}
      <polygon points={areaPoints} fill="#E91E63" fillOpacity="0.08" />
      <polyline points={points} fill="none" stroke="#E91E63" strokeWidth="2.5"
        strokeLinejoin="round" strokeLinecap="round" />
      {data.map((d,i) => (
        <g key={i}>
          <circle cx={toX(i)} cy={toY(d.precio)} r="4" fill="#E91E63" />
          <text x={toX(i)} y={H - 6} textAnchor="middle"
            fontSize="10" fill="var(--text-secondary)">
            {fmtFecha(d.scrapeado_en)}
          </text>
        </g>
      ))}
    </svg>
  );
}

function Paginacion({ paginaActual, totalPaginas, onCambiar }) {
  if (totalPaginas <= 1) return null;
  
  const getVisiblePages = () => {
    const pages = [];
    const maxVisible = 5;
    let start = Math.max(1, paginaActual - Math.floor(maxVisible / 2));
    let end = Math.min(totalPaginas, start + maxVisible - 1);
    
    if (end - start < maxVisible - 1) {
      start = Math.max(1, end - maxVisible + 1);
    }
    
    if (start > 1) pages.push(1);
    if (start > 2) pages.push("...");
    
    for (let i = start; i <= end; i++) pages.push(i);
    
    if (end < totalPaginas - 1) pages.push("...");
    if (end < totalPaginas) pages.push(totalPaginas);
    
    return pages;
  };

  return (
    <div className="pagination">
      <button 
        className="page-btn" 
        onClick={() => onCambiar(paginaActual - 1)}
        disabled={paginaActual === 1}
      >
        ← Anterior
      </button>
      
      {getVisiblePages().map((page, i) => 
        page === "..." ? (
          <span key={i} style={{color:"var(--text-secondary)",padding:"0 4px"}}>...</span>
        ) : (
          <button
            key={page}
            className={`page-btn ${page === paginaActual ? 'active' : ''}`}
            onClick={() => onCambiar(page)}
          >
            {page}
          </button>
        )
      )}
      
      <button 
        className="page-btn" 
        onClick={() => onCambiar(paginaActual + 1)}
        disabled={paginaActual === totalPaginas}
      >
        Siguiente →
      </button>
    </div>
  );
}

function AuthModal({ onClose, onLogin }) {
  const [modo, setModo]         = React.useState("login");
  const [email, setEmail]       = React.useState("");
  const [password, setPassword] = React.useState("");
  const [error, setError]       = React.useState("");
  const [loading, setLoading]   = React.useState(false);
  const [msg, setMsg]           = React.useState("");

  async function handleSubmit() {
    setError(""); setMsg(""); setLoading(true);
    const endpoint = modo === "login" ? "/auth/login" : "/auth/register";
    try {
      const res = await fetch(`${API}${endpoint}`, {
        method: "POST",
        headers: {"Content-Type": "application/json"},
        body: JSON.stringify({email, password})
      });
      const data = await res.json();
      if (!res.ok) throw new Error(data.detail || "Error");
      if (modo === "login") {
        onLogin(data.access_token, data.refresh_token, data.user);
        onClose();
      } else {
        setMsg("¡Registro exitoso! Revisá tu email para confirmar tu cuenta.");
        setModo("login");
      }
    } catch(e) {
      setError(e.message);
    }
    setLoading(false);
  }

  return (
    <div className="modal-overlay" onClick={onClose}>
      <div className="modal" style={{maxWidth:400}} onClick={e => e.stopPropagation()}>
        <div className="modal-header">
          <div className="modal-title">{modo === "login" ? "Iniciar sesión" : "Crear cuenta"}</div>
          <button className="modal-close" onClick={onClose}>✕</button>
        </div>
        <div className="modal-content">
          {msg && <div style={{background:"#E8F5E9",color:"#2E7D32",padding:"12px 16px",borderRadius:10,marginBottom:16,fontSize:13,fontWeight:600}}>{msg}</div>}
          {error && <div style={{background:"#FCE4EC",color:"#C62828",padding:"12px 16px",borderRadius:10,marginBottom:16,fontSize:13}}>{error}</div>}
          <input className="auth-input" type="email" placeholder="Email"
            value={email} onChange={e => setEmail(e.target.value)} 
            style={{width:"100%",padding:"12px 16px",borderRadius:10,border:"2px solid var(--border)",marginBottom:12,fontSize:14,fontFamily:"inherit",outline:"none"}} />
          <input className="auth-input" type="password" placeholder="Contraseña"
            value={password} onChange={e => setPassword(e.target.value)}
            onKeyDown={e => e.key === "Enter" && handleSubmit()}
            style={{width:"100%",padding:"12px 16px",borderRadius:10,border:"2px solid var(--border)",marginBottom:16,fontSize:14,fontFamily:"inherit",outline:"none"}} />
          <button 
            onClick={handleSubmit} 
            disabled={loading}
            style={{width:"100%",padding:"14px",background:"var(--text)",color:"#fff",border:"none",borderRadius:10,fontSize:14,fontWeight:700,cursor:"pointer",fontFamily:"inherit",opacity:loading?0.7:1}}>
            {loading ? "Cargando..." : modo === "login" ? "Iniciar sesión" : "Crear cuenta"}
          </button>
          <div style={{textAlign:"center",marginTop:16}}>
            <button 
              onClick={() => {setModo(modo==="login"?"register":"login"); setError(""); setMsg("");}}
              style={{background:"none",border:"none",color:"var(--text-secondary)",fontSize:13,cursor:"pointer",textDecoration:"underline"}}>
              {modo === "login" ? "¿No tenés cuenta? Registrate" : "¿Ya tenés cuenta? Iniciá sesión"}
            </button>
          </div>
        </div>
      </div>
    </div>
  );
}

function App() {

  const [loading, setLoading]         = React.useState(true);
  const [tab, setTab]                 = React.useState("todos");

  const [catFiltro, setCatFiltro]     = React.useState("Todas");
  const [subcatFiltro, setSubcatFiltro] = React.useState("Todas");
  const [soloOfertas, setSoloOfertas] = React.useState(false);
  const [orden, setOrden]             = React.useState("popular");
  const [busqueda, setBusqueda]       = React.useState("");
  const [modalProducto, setModalProducto] = React.useState(null);
  const [porPagina, setPorPagina]         = React.useState(25);
  const [pagina, setPagina]               = React.useState(1);
  const [totalProductos, setTotalProductos] = React.useState(0);
  const [totalPaginas, setTotalPaginas]     = React.useState(1);
  const [usuario, setUsuario]             = React.useState(() => {
    try { return JSON.parse(localStorage.getItem("sp_user")); } catch { return null; }
  });
  const [token, setToken]                 = React.useState(() => localStorage.getItem("sp_token") || null);
  const [showAuth, setShowAuth]           = React.useState(false);
  const [toast, setToast]                 = React.useState(null);

  function showToast(message, type = 'success') {
    setToast({ message, type });
  }

  function handleLogin(access_token, refresh_token, user) {
    setToken(access_token);
    setUsuario(user);
    localStorage.setItem("sp_token", access_token);
    localStorage.setItem("sp_refresh", refresh_token || "");
    localStorage.setItem("sp_user", JSON.stringify(user));
    showToast(`¡Bienvenido ${user.email}!`);
  }
  
  function handleLogout() {
    setToken(null);
    setUsuario(null);
    localStorage.removeItem("sp_token");
    localStorage.removeItem("sp_refresh");
    localStorage.removeItem("sp_user");
    showToast("Sesión cerrada", "success");
  }

  React.useEffect(() => {
    if (!token) return;
    try {
      const payload = JSON.parse(atob(token.split(".")[1]));
      const expiresIn = payload.exp * 1000 - Date.now() - 5 * 60 * 1000;
      if (expiresIn <= 0) {
        refreshToken();
        return;
      }
      const timer = setTimeout(refreshToken, expiresIn);
      return () => clearTimeout(timer);
    } catch(e) {}
  }, [token]);

  async function refreshToken() {
    const refresh = localStorage.getItem("sp_refresh");
    if (!refresh) { handleLogout(); return; }
    try {
      const res = await fetch(`${API}/auth/refresh`, {
        method: "POST",
        headers: {"Content-Type": "application/json"},
        body: JSON.stringify({refresh_token: refresh})
      });
      if (!res.ok) { handleLogout(); return; }
      const data = await res.json();
      setToken(data.access_token);
      localStorage.setItem("sp_token", data.access_token);
      if (data.refresh_token) localStorage.setItem("sp_refresh", data.refresh_token);
    } catch(e) {
      handleLogout();
    }
  }

  const cargarRef = React.useRef(null);

  async function cargar(paginaOverride) {
    if (cargarRef.current) cargarRef.current.abort();
    const controller = new AbortController();
    cargarRef.current = controller;
    const signal = controller.signal;

    setLoading(true);
    const paginaActual = paginaOverride !== undefined ? paginaOverride : pagina;
    const params = new URLSearchParams();
    if (superFiltro !== "Todos") params.append("supermercado", superFiltro);
    if (catFiltro !== "Todas")   params.append("categoria", catFiltro);
    if (subcatFiltro !== "Todas") params.append("subcategoria", subcatFiltro);
    if (soloOfertas)             params.append("ofertas", "true");
    if (tab === "favoritos")     params.append("favoritos", "true");
    if (busqueda)                params.append("busqueda", busqueda);
    params.append("pagina", paginaActual);
    params.append("por_pagina", porPagina === 0 ? 999999 : porPagina);
    params.append("orden", orden);

    try {
      const headers = token ? {Authorization:`Bearer ${token}`} : {};
      const [resp, st] = await Promise.all([
        fetch(`${API}/productos?${params}`, {signal, headers}).then(r => r.json()),
        fetch(`${API}/stats`, {signal, headers}).then(r => r.json()),
      ]);
      if (signal.aborted) return;
      const prods = resp.productos || [];
      setProductos(Array.isArray(prods) ? prods : []);
      setTotalProductos(resp.total || 0);
      setTotalPaginas(resp.total_paginas || 1);
      setStats(st);
    } catch(e) {
      if (e.name === "AbortError") return;
    }
    setLoading(false);
  }

  React.useEffect(() => {
    setPagina(1);
    const t = setTimeout(() => cargar(1), busqueda ? 400 : 0);
    return () => clearTimeout(t);
  }, [tab, superFiltro, catFiltro, subcatFiltro, soloOfertas, busqueda, token, orden, porPagina]);

  async function toggleFav(e, id) {
    e.stopPropagation();
    if (!token) { setShowAuth(true); return; }
    
    const product = productos.find(p => p.id === id);
    const newFavState = !product.favorito;
    
    setProductos(prev => prev.map(p => p.id === id ? {...p, favorito: !p.favorito} : p));
    
    showToast(newFavState ? "Agregado a favoritos 💚" : "Eliminado de favoritos");
    
    const res = await fetch(`${API}/productos/${id}/favorito`, { 
      method: "POST", 
      headers: token ? {Authorization:`Bearer ${token}`} : {} 
    });
    const data = await res.json().catch(() => ({}));
    
    if (data.favorito !== undefined) {
      setProductos(prev => prev.map(p => p.id === id ? {...p, favorito: data.favorito} : p));
    }
    
    fetch(`${API}/stats`, {headers:{Authorization:`Bearer ${token}`}})
      .then(r => r.json()).then(st => setStats(st)).catch(() => {});
  }

  function cambiarPagina(p) {
    setPagina(p);
    cargar(p);
    window.scrollTo({ top: 0, behavior: "smooth" });
  }
  
  function cambiarPorPagina(val) {
    setPorPagina(val);
    setPagina(1);
  }

  function handleCatFiltro(cat) {
    setCatFiltro(cat);
    setSubcatFiltro("Todas");
  }

  const supers = ["Todos","VEA","Carrefour","MasOnline","Dino"];
  const cats   = ["Todas", ...Object.keys(CATEGORIAS)];
  const subcats = catFiltro !== "Todas" && CATEGORIAS[catFiltro]
    ? ["Todas", ...Object.keys(CATEGORIAS[catFiltro].subs)]
    : [];

  return (
    <div>
      {toast && <Toast message={toast.message} type={toast.type} onClose={() => setToast(null)} />}
      {showAuth && <AuthModal onClose={() => setShowAuth(false)} onLogin={handleLogin} />}
      {modalProducto &&
        <Modal producto={modalProducto} onClose={() => setModalProducto(null)} />}

      <div className="header">
        <div>
          <h1 style={{cursor:"pointer"}} onClick={() => { 
            setTab("todos"); setSuperFiltro("Todos"); setCatFiltro("Todas"); 
            setSubcatFiltro("Todas"); setBusqueda(""); setSoloOfertas(false); 
            window.scrollTo({top:0,behavior:"smooth"}); 
          }}>
            🛒 <span>Super</span>Precios AR
          </h1>
          <p>Compará precios en tiempo real · VEA · Carrefour · MásOnline · Dino</p>
        </div>
        <div style={{display:"flex",alignItems:"center",gap:12}}>
          {usuario ? (
            <div style={{display:"flex",alignItems:"center",gap:8}}>
              <span style={{fontSize:12,color:"#9ca3af"}}>{usuario.email}</span>
              <button onClick={handleLogout}
                style={{fontSize:11,padding:"6px 12px",borderRadius:8,border:"1px solid #374151",
                  background:"none",color:"#9ca3af",cursor:"pointer",transition:"all 0.2s"}}>
                Salir
              </button>
            </div>
          ) : (
            <button onClick={() => setShowAuth(true)}
              style={{fontSize:12,padding:"8px 16px",borderRadius:8,border:"none",
                background:"var(--primary)",color:"#000",fontWeight:700,cursor:"pointer",
                transition:"all 0.2s",boxShadow:"0 2px 8px rgba(255,214,0,0.3)"}}>
              Iniciar sesión
            </button>
          )}
          <div className="stats">
            <div className="stat" onClick={() => { 
              setTab("todos"); setSuperFiltro("Todos"); setCatFiltro("Todas"); 
              setSubcatFiltro("Todas"); setBusqueda(""); setSoloOfertas(false); 
              window.scrollTo({top:0,behavior:"smooth"}); 
            }}>
              <div className="stat-val" style={{color:"var(--primary)"}}>{stats.total_productos||0}</div>
              <div className="stat-lbl">Productos</div>
            </div>
            <div className="stat" onClick={() => { 
              if (!token) { setShowAuth(true); return; } 
              setTab("favoritos"); window.scrollTo({top:0,behavior:"smooth"}); 
            }}>
              <div className="stat-val" style={{color:"#4CAF50"}}>{stats.favoritos||0}</div>
              <div className="stat-lbl">Favoritos</div>
            </div>
          </div>
        </div>
      </div>

      <div className="container">
        <div className="tabs">
          {[["todos","Todos los productos"],["favoritos","Mis favoritos"]].map(([key,label]) => (
            <button key={key} className={`tab ${tab===key?"active":""}`}
              onClick={() => {
                if (key === "favoritos" && !token) { setShowAuth(true); return; }
                setTab(key);
              }}>
              {label}
              {key==="favoritos" && stats.favoritos > 0 &&
                <span style={{marginLeft:8,background:"#E91E63",color:"#fff",
                  borderRadius:10,fontSize:10,padding:"2px 8px",fontWeight:700}}>
                  {stats.favoritos}
                </span>}
            </button>
          ))}
        </div>

        <div className="search">
          <span>🔍</span>
          <input 
            value={busqueda} 
            onChange={e => setBusqueda(e.target.value)}
            placeholder="¿Qué producto estás buscando? Ej: leche, coca, arroz..."
          />
        </div>

        <div className="filters">
          <div className="filter-row">
            <span className="filter-label">Supermercado</span>
            {supers.map(s => {
              const c = SUPER_COLORS[s];
              return (
                <button key={s} className={`pill ${superFiltro===s?"active":""}`}
                  style={superFiltro===s?{background:c?.bg||"#1a1a1a",color:c?.text||"#fff"}:{}}
                  onClick={() => setSuperFiltro(s)}>{s}</button>
              );
            })}
          </div>
          <div className="filter-row">
            <span className="filter-label">Categoría</span>
            {cats.map(c => (
              <button key={c} className={`pill ${catFiltro===c?"active":""}`}
                style={catFiltro===c?{background:"#1a1a1a",color:"#fff"}:{}}
                onClick={() => handleCatFiltro(c)}>
                {CAT_LABELS[c]||"Todas"}
              </button>
            ))}
          </div>
          {subcats.length > 0 && (
            <div className="filter-row">
              <span className="filter-label">Subcategoría</span>
              <select
                value={subcatFiltro}
                onChange={e => setSubcatFiltro(e.target.value)}
                style={{padding:"8px 16px",borderRadius:24,fontSize:12,fontWeight:600,
                  border:"1.5px solid var(--border)",background:"var(--card)",color:"var(--text)",
                  cursor:"pointer",outline:"none",fontFamily:"inherit"}}>
                {subcats.map(s => (
                  <option key={s} value={s}>
                    {s === "Todas" ? "Todas las subcategorías" : SUBCAT_LABELS[s] || s}
                  </option>
                ))}
              </select>
            </div>
          )}
          <div className="filter-row">
            <span className="filter-label">Ordenar por</span>
            {[["popular","Más populares"],["precio-asc","Precio ↑"],
              ["precio-desc","Precio ↓"],["descuento","Mayor descuento"],["reciente","Más reciente"]].map(([key,label]) => (
              <button key={key} className={`pill ${orden===key?"active":""}`}
                style={orden===key?{background:"#1a1a1a",color:"#fff"}:{}}
                onClick={() => setOrden(key)}>{label}</button>
            ))}
          </div>
          <div className="filter-row">
            <span className="filter-label">Ofertas</span>
            <div 
              onClick={() => setSoloOfertas(!soloOfertas)}
              className={`offer-toggle ${soloOfertas ? 'active' : ''}`}>
              <div className="toggle-switch">
                <div className="toggle-knob"></div>
              </div>
              <span style={{fontSize:13,fontWeight:700,color:soloOfertas?"#E91E63":"var(--text-secondary)"}}>
                Solo ofertas y promos
              </span>
            </div>
          </div>
        </div>

        <div className="result-count">
          <span>
            {loading ? "Cargando productos..." : `${totalProductos.toLocaleString()} producto${totalProductos!==1?"s":""} · página ${pagina} de ${totalPaginas}`}
          </span>
          <div style={{display:"flex",gap:8,alignItems:"center",flexWrap:"wrap"}}>
            <span style={{fontSize:11,color:"var(--text-secondary)"}}>Mostrar:</span>
            {[[25,"25"],[50,"50"],[100,"100"],[0,"Todos"]].map(([val,label]) => (
              <button key={val} onClick={() => cambiarPorPagina(val)}
                className={`pill ${porPagina===val?"active":""}`}
                style={{padding:"4px 12px",fontSize:11,
                  ...(porPagina===val?{background:"#1a1a1a",color:"#fff",border:"none"}:{})}}>
                {label}
              </button>
            ))}
          </div>
        </div>

        {loading ? (
          <div className="grid">
            {[...Array(6)].map((_, i) => <SkeletonCard key={i} />)}
          </div>
        ) : productos.length === 0 ? (
          <div className="empty">
            <div>🔍</div>
            <p>No encontramos productos con esos filtros</p>
            <p style={{fontSize:13,marginTop:8,opacity:0.7}}>Probá con otra búsqueda o ajustá los filtros</p>
          </div>
        ) : (
          <>
            <div className="grid">
              {productos.map(p => {
                const hasOffer = p.descuento_pct || p.promocion;
                const sc = SUPER_COLORS[p.supermercado] || {bg:"#888",text:"#fff"};
                const cheapestSuper = p.otros_supers?.length > 0 
                  ? [...p.otros_supers].sort((a,b) => a.precio - b.precio)[0]
                  : null;
                const isCheapest = !cheapestSuper || p.precio <= cheapestSuper.precio;
                
                return (
                  <div key={p.id} className={`card ${hasOffer?"oferta":""}`}
                    onClick={() => setModalProducto(p)}>
                    
                    <div style={{position:"relative"}}>
                      <img src={p.imagen_url || 'data:image/svg+xml,<svg xmlns="http://www.w3.org/2000/svg" width="100" height="100"><rect width="100" height="100" fill="%23f0f0f0"/><text x="50" y="50" text-anchor="middle" fill="%23999" font-size="40">📦</text></svg>'} 
                        alt={p.nombre}
                        className="card-image"
                        loading="lazy"
                      />
                      <span className="super-badge super-float" style={{background:sc.bg,color:sc.text}}>
                        {p.supermercado}
                      </span>
                      <button className="fav-btn" onClick={e => toggleFav(e, p.id)}>
                        {p.favorito ? "❤️" : "🤍"}
                      </button>
                      {p.scrapeado_en && (
                        <span className="time-badge" style={{
                          position:"absolute",bottom:8,right
:8,fontSize:11,color:"var(--text-secondary)",background:"rgba(255,255,255,0.9)",padding:"4px 8px",borderRadius:12,zIndex:2}}>
                          {relTime(p.scrapeado_en)}
                        </span>
                      )}
                    </div>

                    <div className="card-content">
                      <div className="badges">
                        <span className="cat-badge">{SUBCAT_LABELS[p.subcategoria] || CAT_LABELS[p.categoria] || p.categoria}</span>
                        {isCheapest && (
                          <span style={{fontSize:10,fontWeight:800,color:"#2E7D32",background:"#E8F5E9",padding:"2px 8px",borderRadius:10}}>
                            💚 Mejor precio
                          </span>
                        )}
                      </div>
                      
                      <div className="product-name">{p.nombre}</div>
                      
                      <div className="prices">
                        <span className={`price ${hasOffer?"oferta":"normal"}`}>
                          {fmt(p.precio)}
                        </span>
                        {p.precio_original && (
                          <span className="price-original">{fmt(p.precio_original)}</span>
                        )}
                        {p.descuento_pct && !p.promocion && (
                          <span className="discount">-{p.descuento_pct}%</span>
                        )}
                      </div>

                      {p.promocion && (
                        <span className="promo" style={{background:promoColor(p.promocion)}}>
                          {p.promocion}
                        </span>
                      )}

                      {p.historial_precios && p.historial_precios.length > 1 && (
                        <Sparkline data={p.historial_precios.slice(-5)} />
                      )}

                      {p.otros_supers && p.otros_supers.length > 0 && (
                        <div className="price-compare">
                          {[...p.otros_supers]
                            .sort((a,b) => a.precio - b.precio)
                            .slice(0, 2)
                            .map((o, i) => {
                              const osc = SUPER_COLORS[o.supermercado] || {bg:"#888",text:"#fff"};
                              const esMasBarato = o.precio < p.precio;
                              return (
                                <a key={o.supermercado} href={o.url} target="_blank" rel="noopener noreferrer"
                                  onClick={e => e.stopPropagation()}
                                  className={`compare-row ${esMasBarato ? 'best' : ''}`}>
                                  <span className="compare-super">
                                    <span style={{fontSize:10,fontWeight:700,color:osc.bg}}>
                                      {o.supermercado}
                                    </span>
                                    {esMasBarato && <span className="best-badge">MEJOR</span>}
                                  </span>
                                  <span style={{fontWeight:700,color:esMasBarato?"#2E7D32":"var(--text)"}}>
                                    {fmt(o.precio)}
                                  </span>
                                </a>
                              );
                            })}
                          {p.otros_supers.length > 2 && (
                            <div style={{textAlign:"center",fontSize:11,color:"var(--text-secondary)",marginTop:4}}>
                              +{p.otros_supers.length - 2} supermercados más
                            </div>
                          )}
                        </div>
                      )}
                    </div>
                  </div>
                );
              })}
            </div>

            <Paginacion
              paginaActual={pagina}
              totalPaginas={totalPaginas}
              onCambiar={cambiarPagina}
            />
          </>
        )}

        <div className="footer">
          <div style={{display:"flex",alignItems:"center",justifyContent:"center",gap:16,marginBottom:12,flexWrap:"wrap"}}>
            <span style={{display:"flex",alignItems:"center",gap:6}}>
              <span style={{fontSize:16}}>🤖</span>
              <span>Actualización automática: 08:00 y 20:00 hs</span>
            </span>
            <span style={{color:"var(--border)"}}>|</span>
            <span style={{display:"flex",alignItems:"center",gap:6}}>
              <span style={{fontSize:16}}>💚</span>
              <span>Agregá favoritos para alertas de precio</span>
            </span>
            <span style={{color:"var(--border)"}}>|</span>
            <span style={{display:"flex",alignItems:"center",gap:6}}>
              <span style={{fontSize:16}}>📊</span>
              <span>Clickeá para ver historial</span>
            </span>
          </div>
          <p style={{fontSize:11,opacity:0.7}}>
            Compará precios entre VEA, Carrefour, MásOnline y Dino · 
            {usuario ? ` Conectado como ${usuario.email}` : ' Iniciá sesión para guardar favoritos'}
          </p>
        </div>
      </div>
    </div>
  );
}

ReactDOM.createRoot(document.getElementById("root")).render(<App />);
